const
phaseshift:real=3.14159;
var
timeslice,angle,wave1,wave2,vector1,vector2,sum,i,j:real;

ClrScr;
grDriver:=Detect;

begin
angle:=0;

begin
wave1:=cos(timeslice+angle);
