Code:
  1. 006F5224   . 53             PUSH EBX
  2. 006F5225   . 8BD8           MOV EBX,EAX
  3. 006F5227   . 33C9           XOR ECX,ECX
  4. 006F5229   . 8B53 40        MOV EDX,DWORD PTR DS:[EBX+40]
  5. 006F522C   . 33C0           XOR EAX,EAX
  6. 006F522E   . E8 0D000000    CALL rutserv.006F5240
  7. 006F5233   . 8943 44        MOV DWORD PTR DS:[EBX+44],EAX <---- Результат выполнения процедуры запоминаем
  8. 006F5236   . 8BC3           MOV EAX,EBX
  9. 006F5238   . E8 9BA9D6FF    CALL rutserv.0045FBD8
  10. 006F523D   . 5B             POP EBX
