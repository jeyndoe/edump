/*
Проверка программы по алгоритму CRC32 от модификаций вирусов (или защита от "дурака")
Вставляется непосредственно в код самой программы
Число CRC записывается в скомпилированный файл с помощью внешней программы
Ее вы можете найти на oldskool.altnet.ru/UpdateCRC32.exe или сделать сами
на основе этих исходников ;)
Тестовая программа лежит здесь: oldskool.altnet.ru/CRC32Test.exe

Идея была навеяна защитой программы SourceFormatX 2.56

Функцию Crc32Prim можно достать из библиотеки SysTools от TurboPower Software
или еще вытащить из zlib (тогда придется кое-что подправить)
Главное, чтобы само число CRC было 32-битным, т.е. занимало в памяти 4 байта

(c) Tim (Dmitry Titkov), 2005
e-mail: cat87@fromru.com
*/


// -----------------
#include <stdio.h>
#include <stcrc.hpp> // Из библиотеки SysTools от TurboPower Software
// -----------------

// ---------------------------------------------------------------------
USEUNIT("C:\Program Files\Borland\CBuilder5\Source\SysTools\stcrc.pas");
// ---------------------------------------------------------------------

long filesize(FILE *stream)
{
long curpos;
long length;
curpos = ftell(stream);
fseek(stream, 0L, SEEK_END);
length = ftell(stream);
fseek(stream, curpos, SEEK_SET);
return length;
}

int size_of_header(FILE *exe)
{
long curpos;
DWORD size;
curpos = ftell(exe);
fseek(exe, 0x254, SEEK_SET); // Size of header, смотрите в Hiew, если интересно ;)
fread(&size, sizeof(size), 1, exe);
fseek(exe, curpos, SEEK_SET);
return size;
}

int read_crc32(FILE *exe)
{
long curpos;
int crc;
curpos = ftell(exe);
fseek(exe, 0x80, SEEK_SET); // число CRC лежит на оффсете 0x80 от начала файла
fread(&crc, sizeof(crc), 1, exe);
fseek(exe, curpos, SEEK_SET);
return crc;
}

bool check_crc32(char *exe_name)
{
int crc;
int read_crc;
BYTE *buf;
FILE *exe = fopen(exe_name, "rb");
DWORD size = size_of_header(exe);
long exe_size = filesize(exe);
fseek(exe, size, SEEK_SET);
buf = new BYTE[exe_size-size+1];
fread(buf, exe_size-size, 1, exe);
crc = Crc32Prim(buf, exe_size-size, 0xFFFFFFFF);
read_crc = read_crc32(exe);
fclose(exe);
if(crc == read_crc)
return 1;
else
return 0;
}

// Эта проверка вставляется в самое начало процедуры WinMain
// Если компилируете не в Borland C++, то поменяйте ParamStr() на другую
// функцию, которая вернет вам имя вашего exe-файла
// ---------------------------------------------------------------------- ----
if(!check_crc32(ParamStr(0).c_str()))
{
Application->MessageBoxA("This file has been modified. Reason might be a possible virus infection!\nPlease scan your system with Anti-Virus software and reinstall the program.", "Protection Error", MB_OK | MB_ICONERROR);
exit(1);
}
// ---------------------------------------------------------------------- ----

unsigned long CalcCRC32(BYTE *buf, long length)
{
  unsigned long CRC32 = 0xFFFFFFFF, tt;
  for (int i = 0; i < length; i++)
  {
    tt = buf[i];
    CRC32 = (CRC32 >> 8) ^ CRC32Table[tt ^ (CRC32 &0x000000FF)];
  }
  return CRC32;
}
