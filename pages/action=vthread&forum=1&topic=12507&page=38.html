static getdumpx(name, ea, len, ofs)
{
auto hin, bb, i;

hin = fopen(name, "rb");
if(hin == 0) {
Message("Can't open in file\n");
return;
}
if(fseek(hin, ofs, 0)!=0) {
Message("Bad seek pointer\n");
fclose(hin);
}
for(i=0; i<len; i=i+1) {
bb = fgetc(hin);
PatchByte(ea+i,bb);
}
fclose(hin);
}
