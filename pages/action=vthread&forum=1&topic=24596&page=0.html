00. Закидываем сайт, в доменной зоне onion.
Нам понадобится,один из дистрибутивов линукс,
желательно Debian, опять-же это дело вкуса.
Нам понадобится ,веб сервер Апач и Тор.
Открываем терминал.

Code:
  1. 01.root@admin: #  apt-get   install  apache2 /  нежно выбиваем установку апач. 
  2.  
  3. 02. root@admin: # service   apache2  restart   проверяем наш апач, вбей в браузер  http://localhost или http://127.0.0.1 
  4. Если что-то нам нужно подредактировать  , или закинуть свой  сайт , наутилус нас погрузит на эту глубину. root@admin: # nautilus  .Наша беда, будет вот поэтому пути. /var/www/html 
  5. Если у вас статические html файлы с сайтом,идеально для безопасности, и не много ресурсов.
  6.  
  7. 03.root@admin: # apt-get   install  tor //  устанавливаем тор .  
  8.  
  9. 04. root@admin: # service  tor restart //проверяем наши tor,все ок! поехали дальше.  
  10.  
  11. 05.root@admin: # cd /var/lib/tor   //залезли в tor .
  12.  
  13. 06. root@admin: /var/lib/tor # nano  /etc/tor/torrc  //залазим в настройки / ищем две строчки.
  14. HiddenServiceDir /var/lib/tor/hidden_service/
  15. HiddenServicePort 80 127.0.0.1:80
  16. рас комментируем их (убираем решетку впереди)сохраняем ctrl+x.
  17.  
  18. 07. root@admin: /var/lib/tor # pwd    /проверяем 1.
  19. /var/lib/tor
  20.  
  21. 08. root@admin: /var/lib/tor # ls     /проверяем 2.
  22. cached-certs     cached-microdesc-consensus cached-microdescs cached-microdescs.new  lock  state  //ок работает .
  23.  
  24. 09. root@admin: # service  tor restart /проверяем наши tor,все ок.
  25.  
  26. 10. root@admin: /var/lib/tor # ls  /проверяем 3.
  27. cached-certs     cached-microdesc-consensus cached-microdescs cached-microdescs.new  hidden_service lock state  // видим hidden_service .
  28.  
  29. 11. root@admin: /var/lib/tor # cd hidden_service // заходим в hidden_service 
  30.  
  31. 12. root@admin: /var/lib/tor # ls  // Первая строчка в конфиге указывает путь к закрытому ключу, который создаётся автоматически TOR-ом при первом запуске после модификации конфига и играет роль onion-домена.
  32.  
  33. hostname  private_key
  34.  
  35. 13. root@admin: /var/lib/tor # cat hostname // Файл hostname содержит адрес домена сгенерированного на базе хеша ключа private_key, который в свою очередь сгенерировался случайным образом при первом запуске.
  36.  
  37. qigf5kfufjz77s5z.onion   // Копируйте, и вставить в браузер тор.
  38.  
  39. 14. /var/lib/tor/  //папка где  находится ключи
