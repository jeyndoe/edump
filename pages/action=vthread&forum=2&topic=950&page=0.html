IF .NOT. ALLTRIM(SYS(2007, ;

IF .NOT. MESSAGEBOX('Повторить попытку?',036,'Неверный пароль')=6
thisform.release

for i=1 to 99999999
IF ALLTRIM(SYS(2007, ;
