1. качаем http://www.mono-project.com/Cecil http://www.mono-project.com/Cecil

3. запускаем:
Code:

  18.         static void Clear(string path)
  19.         {
  20.             var asm = AssemblyFactory.GetAssembly(path);
  22.             for (int i = 0; i < asm.CustomAttributes.Count; ++i)
  23.             {
  24.                 var atr = asm.CustomAttributes[i];
  26.                 if (atr.Constructor.DeclaringType.Name == "SuppressIldasmAttribute")
  27.                 {
  28.                     asm.CustomAttributes.Remove(atr);
