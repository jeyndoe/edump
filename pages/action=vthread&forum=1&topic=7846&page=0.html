function crypt_md5_bin($x)
{
$x = md5($x);
$r = ''; $n = strlen($x);
for($i=0;$i<$n;$i=$i+8) $r .= pack('N',hexdec(substr($x,$i,8)));
return $r;
}

function crypt_data($data,$key)
{
$res = '';
$len = strlen($key);
$c = strlen($data);
for($i=0;$i<$c;$i=$i+$len)
{
$res .= $key ^ substr($data,$i,$len);
$key = crypt_md5_bin($key);
}
return $res;
