Ничего особенного, стандартный манифест для UAC:
Code:
  1. <assembly xmlns="urn:schemas-microsoft-com:asm.v1" manifestVersion="1.0">
  2.          <trustInfo xmlns="urn:schemas-microsoft-com:asm.v3">
  3.                  <security>
  4.                         <requestedPrivileges>
  5.                               <requestedExecutionLevel level="asInvoker" uiAccess="false">
  6.                               </requestedExecutionLevel>
  7.                         </requestedPrivileges>
  8.                  </security>
  9.          </trustInfo>
  10.          <application xmlns="urn:schemas-microsoft-com:asm.v3">
  11.                  <windowsSettings>
  12.                         <ms_windowsSettings:dpiAware xmlns:ms_windowsSettings="http://schemas.microsoft.com/SMI/2005/WindowsSettings" xmlns="http://schemas.microsoft.com/SMI/2005/WindowsSettings">true</ms_windowsSettings:dpiAware>
  13.                  </windowsSettings>
  14.          </application>
  15. </assembly>
