Добавить вызов любой API функции из kernel32.dll (и включить соответствующие инклуды) в любое место кода.
Примерно так:
;PROG1.ASM:
.586P
.MODEL FLAT, STDCALL
include \masm32\include\kernel32.inc
includelib \masm32\lib\kernel32.lib
EXTERN PROC1@0:NEAR
.code
START:
CALL PROC1@0
RET
invoke GetCommandLine
END START


;PROG2.ASM:
.586P
.MODEL FLAT, STDCALL
PUBLIC PROC1
.code
PROC1 PROC
MOV EAX, 1000
RET
PROC1 ENDP
END
