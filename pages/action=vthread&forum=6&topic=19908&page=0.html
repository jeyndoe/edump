Code:
  1. int __fastcall BinSearch(char *buf, int size, char *str, int len)
  2. {
  3. int i, poz;
  4.  
  5.          for (i = 0; i <= size; i++)
  6.          {
  7.                  if (buf[i] == str[0])
  8.                  {
  9.                         for (poz = 1; poz < len; poz++)
  10.                         {
  11.                               if (buf[i+poz] != str[poz]) break;
  12.                               if (poz == len-1) return i;
  13.                         }
  14.                  }
  15.          }
  16.          return -1;
  17. }
