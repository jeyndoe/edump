Private Sub CommandButton1_Click()
Dim name As String, lname As Integer, serial As String
name = TextBox1.Text: lname = Len(name): errnum = 0

If Len(name) < 5 Then
errnum = 1
GoTo name_err
End If

For i = 1 To lname
a = Asc(Mid(name, i, 1))
If a > 200 Then
errnum = 2
GoTo name_err
End If
b = 200 - a - i
c = c & CStr(b)
Next i

Select Case (lname - 5)
Case 0: serial = c & "T"
Case 1: serial = c & "R"
Case 2: serial = c & "I"
Case 3: serial = c & "P"
Case 4: serial = c & "L"
Case Else: serial = c & "E"
End Select

TextBox2.Text = serial
Exit Sub

name_err:
If errnum = 1 Then TextBox2.Text = "Enter a longer name..."
If errnum = 2 Then TextBox2.Text = "Invalid name entered!!!"
End Sub
