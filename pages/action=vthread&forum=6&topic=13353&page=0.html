Находишь окно с помощью, (предварительно узнав имя/класс через Spy++)
Code:
  2. ; Находим окно
  3. push szClassWindow
  4. push szWindowName
  5. call FindWindow
  6. mov hWnd,eax
  7. cmp eax,0
  8. jz error
  10. ; Получаем доступ к его меню
  11. push hWnd
  12. call GetMenu
  13. mov hMenu,eax
  14. cmp eax,0
  15. jz error
  17. ; Теперь можешь: добавлять/удалять/пересчитать пункты меню
  19. push hWnd
  20. push hMenu
  21. call SetMenu
  22. cmp eax,0
  23. jz error
  25. ; сообщение о успешном редактировании меню
  26. push 0
  27. push szText
  28. push szMsg
  29. push 0
  30. call MessageBox
  32. ; При ошибки в регистре EAX=0
  33. error: 
  34. xor eax,eax
  35. ret


О функциях для работы с меню и против ;)
