  6. -0002 ; Frame size: 2; Saved regs: 2; Purge: 0
  7. -0002 ;
  8. -0002
  9. -0002 var_2           db ?
  10. -0001                 db ? ; undefined
  11. +0000  s              db 2 dup(?)
  12. +0002  r              db 4 dup(?)

  15. +0007 ; end of stack variables
