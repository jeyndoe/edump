Code:
  1. .text:0044E49C                 db 3Dh
  2. .text:0044E49D ; ---------------------------------------------------------------------- -----
  3. .text:0044E49D                 push    ebp
  4. .text:0044E49E                 mov     ebp, esp

Должно было быть так:
Code:
  1. .text:0044E49C                 cmp     eax, 0EC8B55h
