Code:
  1.    mov ax,2B
  2.    mov fs,ax
  3. L: mov ax,fs
  4.    cmp ax,2B
  5.    je L
  6. ; swapcontext..

Code:
  1.    mov rcx,[rbp-10]
  2.    mov rdx,[rbp-18]
  3.    push [rbp-20]
