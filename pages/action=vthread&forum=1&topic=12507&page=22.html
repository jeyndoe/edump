static dump(name, ea, len)
{
auto hout, bb, i;

// open output file
hout = fopen(name, "wb");
if(hout == 0) {
Message("Can't open out file\n");
return;
}
if(fseek(hout, 0, 0)!=0) { // from begin file
Message("Bad seek pointer\n");
fclose(hout); // close file
}
for(i=0; i<len; i=i+1) {
bb = Byte(ea+i);
// write byte
fputc(bb, hout);
}
fclose(hout); // close file
}
