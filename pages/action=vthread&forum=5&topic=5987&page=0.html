Я нашел как ОНО переделывает числа. На АСМЕ примерно так:

.586p
.model flat, stdcall
option casemap: none

include \masm32\include\windows.inc
include \masm32\include\masm32.inc
include \masm32\include\kernel32.inc
include \masm32\macros\macros.asm
includelib \masm32\lib\masm32.lib
includelib \masm32\lib\kernel32.lib

EXTERN ExitProcess@4:NEAR


.data
cap db 'Во что переделало:', 0
MyStr db '123456',0
En dd 1Ah
temp dd 0
count db 6

.code
start:
xor ebx, ebx
lea edi, MyStr
mov bl, byte ptr[edi]
inc edi

_loop:
movsx ecx, bl
sub ecx, 30h
mov esi, temp
imul esi, En
add esi, ecx
mov temp, esi

mov bl, byte ptr[edi]
inc edi
dec count
cmp count, 0
jnz _loop

push 0
call ExitProcess@4
end start

В temp переведенное число и хранится...

Так вот теперь надо сделать так, чтобы одна строка - вторая строка - третья строка = определенное число...
