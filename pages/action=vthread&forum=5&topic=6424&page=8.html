"KDExtensions"="IceExt.SYS"

[HKEY_LOCAL_MACHINE\SYSTEM\ControlSet002\Services\NTice]
"KDExtensions"="IceExt.SYS"

2.
REGEDIT4

[HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Services\NTice]
"KDExtensions"=""
