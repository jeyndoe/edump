function running_inside_vpc: boolean; assembler;
asm
push ebp

mov ecx, offset @@exception_handler
mov ebp, esp

push ebx
push ecx
push dword ptr fs:[0]
mov dword ptr fs:[0], esp

mov ebx, 0
mov eax, 1

db 00Fh, 03Fh, 007h, 00Bh
