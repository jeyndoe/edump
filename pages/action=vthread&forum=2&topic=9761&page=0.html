//******************************************************
// Запускать стоя на OEP


var startscan
var OEP
var temp
var TempESP

mov OEP, eip
mov TempESP, ESP
mov startscan, 0053C000 // Начало IAT
rec:
cmp startscan, 0053CC54 // конец IAT?
je endscript
mov temp,[startscan]
cmp temp, 00000000 // Проверка разделителя между библами
je next
cmp temp,10000000 // Есть немного неиспорченых ф-й
ja next
mov eip, temp
add temp, 17
bp temp
run
bc temp
mov [startscan], eax
next:
add startscan, 4
jmp rec
endscript:
mov eip, OEP
mov ESP, TempESP
ret
//******************************************************
