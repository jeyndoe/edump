  7. ;Windows 2k/XP/2k3

  9. [HKEY_LOCAL_MACHINE\SYSTEM\CurrentControlSet\Control\Nls\CodePage]
  10. "1250"="c_1251.nls"
  11. "1251"="c_1251.nls"
  12. "1252"="c_1251.nls"
