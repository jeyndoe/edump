На адобовских продуктах стоит SafeCast, а не SD (просто к сведенью). Им защищена AdobeLM.dll (в 7-ом акробате и 12-м иллюстраторе) и Support.dll, так что exe-шник тут нипричём ;) Я распаковывал AdobeLM.dll в Acrobat Professional 7.0.1. До OEP можно дойти по bpm esp-4. Если у кого-то SafeCast палит дебаггер, то попробуйте заюзать такой скрипт (все эксепшены заигнорить)
// Anti debugger checking for SafeCast2/SafeDisc2
// Author: sanniassin

gpa "IsDebuggerPresent","kernel32.dll"
go $RESULT
rtr
mov eax,0
sto
rtr
sto
sto
add eip,11C
rtr
mov eax,0
ret

Для восстановления импорта я юзаю такой скрипт
// Import recovering for SafeCast2 and maybe SafeDisc2
// Author: sanniassin

var iat_addr
mov iat_addr,10040000 //начало IAT
var x
var y
var check_patched

qqq:
mov x,[iat_addr]
mov y,x
add y,5
mov y,[y]
cmp y,6854609C
jne next_addr
mov eip,x
findop eip,#E8????????#
go $RESULT
sti
cmp check_patched,0
jne already_patched
findop eip,#740E#
mov [$RESULT],#EB#
inc check_patched
already_patched:
find eip,#899401????????FF75F4FF75D4FF75ECFF75E4E8????????83C410#
add $RESULT,18
go $RESULT
mov [iat_addr],ecx
next_addr:
mov esp,6FFC4 //Если будут глюки, то можно убрать
add iat_addr,4
cmp iat_addr,10040380 //Конец IAT
je end
cmp [iat_addr],0
jne qqq
add iat_addr,4
jmp qqq
end:
ret

Там ещё возможно будут спёртые комманды, но они легко находятся и легко фиксятся (SafeCast сам их восстанавливает после 1-го выполнения)
