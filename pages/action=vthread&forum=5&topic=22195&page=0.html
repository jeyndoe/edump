  1. template <typename T>
  2. T modpow(T base, T exp, T modulus)
  3. {
  4.     base %= modulus;
  5.     T result = 1;
  6.     while (exp > 0)
  7.     {
  8.         if (exp & 1) result = (result * base) % modulus;
  9.         base = (base * base) % modulus;
  10.         exp >>= 1;
  11.     }
  12.     return result;
  13. }
