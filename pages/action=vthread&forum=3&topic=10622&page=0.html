Extract WaReZ files
================
mkdir Extracted
cd Extracted
mkdir rar
cd rar
pkzip25 -ext -dir -over=all -nozip -mask=none -times=mod ../../%n
cd ..
for %n in (rar/*.rar) do rar x -y -c- rar/%n
rem del /F /Q rar/*.*
del /F /Q /S rar
rmdir rar
