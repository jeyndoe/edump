Нашел и испытал способ использовать IL прямо в исходниках C# да и любого другого .net языка. VS2012 хавает этот код безо всяких проблем. Но есть одно но, где же взять таблетку от лени... ))) Что бы закончить это нужно где то 10 часов. Короче если наберется пара человек кому это пригодится я закончу.

Вот даже покажу пример как это выглядит (исходник на C# со вставками MSIL):

Code:
  1.         static void Test_UsingVariables()
  2.         {
  3.             Ext.AddVariable("System.Int32"); // 1
  4.             Ext.AddVariable("System.Int32"); // 2
  5.             Ext.AddVariable("System.Int32"); // 3
  6.             Ext.AddVariable("System.Int32"); // 4
  7.             Ext.AddVariable("System.Int32"); // 5
  8.             Ext.AddVariable("System.Int32", "bill"); // 6 bill
  9.  
  10.             // #0 Note. The variables declared in C#, declare first. So here the variable 'mark' has 0 index.
  11.             int mark = 0; // 0 mark
  12.             
  13.             // #1 Note. The C# compiler removes the variable names in the release mode. If you decide to apply 
  14.             //          variable name, you must explicitly specify a name.
  15.             Ext.SetVariableName(0, "mark");
  16.  
  17.             // Init variables
  18.             IL.Ldc_I4_S(10); // Load 10 on stack
  19.             IL.Stloc(mark); // Load variable mark from stack. Analogs IL.Stloc("mark"); and IL.Stloc_0();
  20.  
  21.             IL.Ldc_I4_S(20); // Load 20 on stack
  22.             IL.Stloc_1(); // Load variable 1 from stack
  23.  
  24.             IL.Ldc_I4_S(30); // Load 30 on stack
  25.             IL.Stloc_2(); // Load variable 2 from stack
  26.  
  27.             IL.Ldc_I4_S(40); // Load 40 on stack
  28.             IL.Stloc_3(); // Load variable 3 from stack
  29.  
  30.             IL.Ldc_I4_S(50); // Load 50 on stack
  31.             IL.Stloc(4); // Load variable 4 from stack
  32.  
  33.             IL.Ldc_I4_S(60); // Load 60 on stack
  34.             IL.Stloc(5); // Load variable 5 from stack
  35.  
  36.             IL.Ldc_I4_S(70); // Load 70 on stack
  37.             IL.Stloc("bill"); // Load variable bill from stack
  38.  
  39.             // Compute summ
  40.             IL.Ldloc("mark"); // Analogs IL.Ldloc(mark); and IL.Ldloc_0();
  41.             IL.Ldloc_1();
  42.             IL.Add();
  43.             IL.Ldloc_2();
  44.             IL.Add();
  45.             IL.Ldloc_3();
  46.             IL.Add();
  47.             IL.Ldloc(4);
  48.             IL.Add();
  49.             IL.Ldloc(5);
  50.             IL.Add();
  51.             IL.Ldloc("bill");
  52.             IL.Add();
  53.  
  54.             // Print result
  55.             IL.Call("System.Console::WriteLine(System.Int32)");
  56.  
  57.             // This function on C# :
  58.

  61.             //int b     = 30;
