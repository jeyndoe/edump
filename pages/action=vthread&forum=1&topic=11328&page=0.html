Visual Protect очень слабый прот скрипт для анпака
var oep
var gp
ask "Enter address code section vp.dll"
cmp $RESULT,0
je quit

mov bdl,$RESULT
GMEMI bdl, MEMORYBASE
mov bdl,$RESULT
find bdl,#E8????????FF65FC#
cmp $RESULT,0
je quit
mov oep,$RESULT+5

find bdl,#8A1A428BF381E6FF0000008BCE8D45F8#
cmp $RESULT,0
je quit
mov gp,$RESULT+33
asm gp, "Call GetProcAddress"
bp oep
erun
sti
cmt eip,"<--oep"
msg " Oep Faund Iat fix"
ret
quit:
msg "Not Visage"
ret
в принципе можно сделать скрипт и без импрека иат в обшем то целая
