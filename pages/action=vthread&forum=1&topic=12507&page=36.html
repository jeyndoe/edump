  3.     while (BADADDR == DfirstB(ptr)) {
  4.         if (Word(ptr - 2) == 0) break;
  5.         ptr = ptr - 4;
  6.     }
  7.     auto type = GetStringType(ptr);

  12.     auto bsize = (len + 1) * 2;
