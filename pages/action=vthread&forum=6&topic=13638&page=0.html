Code:
  2. function RVAToOffset(Header: PIMAGENTHEADERS;
  3.                     FileMap: pointer;
  4.                     RVA: dword):dword;
  5. var
  6.   Sec: PIMAGESECTIONHEADER;
  7. begin
  8.   //ImageRvaToSection укажет на секцию, которой принадлежит RVA.
  9.   Sec:=ImageRvaToSection(Header, FileMap, RVA);
  10.   Result:=RVA - Sec.VirtualAddress + Sec.PointerToRawData;
  11. end;
