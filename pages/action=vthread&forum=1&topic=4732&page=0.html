00405329 87D2 XCHG EDX,EDX ; kernel32.77E4F38C
0040532B 9B WAIT
0040532C 90 NOP

00401000 E8 FFFFFFFF CALL keygenme.00401004
00401005 C05D 83 ED RCR [BYTE SS:EBP-7D],0ED ; Shift constant out of range 1..31
00401009 05 64A13000 ADD EAX,30A164
