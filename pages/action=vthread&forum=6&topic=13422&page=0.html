_dword2bin proc near
push eax
push edi
push ebx

mov bh, 32
_convert:
shl eax, 1
mov bl, 30h
adc bl, 0

mov byte ptr[edi], bl

inc edi
dec bh
jnz _convert

mov byte ptr [edi], 0

pop ebx
pop edi
pop eax
ret

_dword2bin endp
