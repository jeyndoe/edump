  6.   szDLLPath db "C:\Windows\Temp\MyDLL.dll", 0

  8.  
  9. start:
  10.     invoke CreateFile, szDLLPath, GENERIC_WRITE, NULL, NULL, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL

  15. End_:
  16.     ret
