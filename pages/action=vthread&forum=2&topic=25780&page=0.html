  1. ##############################################################################

  3. # Name: hello_world_plugin.py
  4. # Auth: @cmatthewbrooks
  5. # Desc: A test plugin to learn how to make these work; Specifically, how to
  6. #       have multiple actions within the same plugin.
  7. #
  8. #       In plain English, IDA will look for the PLUGIN_ENTRY function which
  9. #       should return a plugin object. This object can contain all the 
  10. #       functionality itself, or it can have multiple actions.
  11. #
  12. #       To register actions, use the idaaapi.register_action() function with
  13. #       an argument of type idaapi.action_desc_t.
  14. #
  15. #       A helpful hint from IDA support, don't register actions inside the
  16. #       Edit/Plugins sub-menu. It's dynamic and can be wiped out and built
  17. #       within a session. It's better to register actions' menu locations
  18. #       elsewhere.
  19. #
  20. #
  21. # With help from:
  22. # 
  23. # http://www.hexblog.com/?p=886
  24. # https://github.com/fireeye/flare-ida/blob/master/plugins/
  25. # https://github.com/mwrlabs/win_driver_plugin/
  26. #
  27. ##############################################################################
  28.  
  29.  
  30.  
  31. import idaapi
  32.  
  33.  
  34.  
  35. # Define callbacks which are the actions I actually
  36. # want to perform
  37.  
  38. def hello_mars():
  39.  
  40.     print('Hello Mars!')
  41.  
  42. def hello_earth():
  43.  
  44.     print('Hello Earth!')
  45.  
  46.  
  47. # Define the action_handler_t object that fires the
  48. # callback function when each action is activated
  49.  
  50. class ActionHandler(idaapi.action_handler_t):
  51.  
  52.     def __init__(self, callback):
  53.         
  54.         idaapi.action_handler_t.__init__(self)
  55.         self.callback = callback
  56.     
  57.     def activate(self, ctx):
  58.  
  59.         self.callback()
  60.         return 1
  61.  
  62.     def update(self, ctx):
  63.         
  64.         return idaapi.AST_ENABLE_ALWAYS
  65.  
  66.  
  67. # Define a method to register all the actions when
  68. # the plugin is initialized
  69.  
  70. def register_actions():   
  71.  
  72.     actions = [
  73.         {
  74.             'id': 'hello:earth',
  75.             'name': 'Hello Earth',
  76.             'hotkey': 'Ctrl+Alt+E',
  77.             'comment': 'Print Hello Earth',
  78.             'callback': hello_earth,
  79.             'menu_location': 'Edit/Hello World/Hello Earth'
  80.         },
  81.         {
  82.             'id': 'hello:mars',
  83.             'name': 'Hello Mars',
  84.             'hotkey': 'Ctrl+Alt+M',
  85.             'comment': 'Print Hello Mars',
  86.             'callback': hello_mars,
  87.             'menu_location': 'Edit/Hello World/Hello Mars'
  88.         }
  89.     ]
  90.  
  91.  
  92.     for action in actions:
  93.  
  94.         if not idaapi.register_action(idaapi.action_desc_t(
  95.             action['id'], # Must be the unique item
  96.             action['name'], # The name the user sees
  97.             ActionHandler(action['callback']), # The function to call
  98.             action['hotkey'], # A shortcut, if any (optional)
  99.             action['comment'] # A comment, if any (optional)
  100.         )):
  101.  
  102.             print('Failed to register ' + action['id'])
  103.  
  104.         if not idaapi.attach_action_to_menu(
  105.             action['menu_location'], # The menu location
  106.             action['id'], # The unique function ID
  107.             0):
  108.  
  109.             print('Failed to attach to menu '+ action['id'])
  110.  
  111.  
  112. # Define the plugin class itself which is returned by
  113. # the PLUGIN_ENTRY method that scriptable plugins use
  114. # to be recognized within IDA
  115.  
  116. class HelloWorldPlugin(idaapi.plugin_t):
  117.  
  118.     # Use the HIDE flag to avoid the entry in 
  119.     # Edit/Plugins since this plugin's run()
  120.     # method has no functionality...it's all
  121.     # in the actions.
  122.  
  123.     flags = idaapi.PLUGIN_HIDE
  124.     comment = 'A test plugin'
  125.     help = 'No help - this is just a test'
  126.     wanted_name = 'Hello World'
  127.     wanted_hotkey = ''
  128.  
  129.     def init(self):
  130.         print('HelloWorldPlugin init')
  131.         
  132.         register_actions()
  133.  
  134.         # Return KEEP instead of OK to keep the
  135.         # plugin loaded since it registers
  136.         # callback actions and hotkeys
  137.         #
  138.         # Use OK if the functionality is 
  139.         # all in the plugin and it does
  140.         # one thing then completes.
  141.         return idaapi.PLUGIN_KEEP
  142.  
  143.     def run(self, arg):
  144.         print('HelloWorldPlugin run')
  145.  
  146.     def term(self):
  147.         print('HelloWorldPlugin term')
  148.  
  149.  
  150. # The PLUGIN_ENTRY method is what IDA calls when
  151. # scriptable plugins are loaded. It needs to
  152. # return a plugin of type idaapi.plugin_t
  153.  
  154. def PLUGIN_ENTRY():
  155.     
  156.     try:
  157.         return HelloWorldPlugin()
  158.     
  159.     except Exception, err:
  160.         import traceback
  161.         print('Error: %s\n%s' % str((err), traceback.format_exc()))
  162.         raise
  163.
