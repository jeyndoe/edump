open "file1.exe" for binary as #1
open "file2.exe" for binary as #2

for i=1 to lof(1)
  get #1,i,byte1
  get #2,i,byte2
  if byte1<>byte2 then sDifferences=sDifferences & vbcrlf & hex(i) & ": " & hex(byte1) & " " & hex(byte1)
next

close

msgbox "Различия: " & vbcrlf & sDifferences
